Using UnifyList and UnifyList<T>

The UnifyList<T> has been designed to be interchangable with the .NET List<T> through the implementation of the IList<T> interface. Unify also provides a non-generic UnifyList, which implements the .NET IList interface.

To create an instance of a UnifyList or UnifyList<T>, use the new() operator and constructor in the same way as a native .NET List<T> constructor; Unify List constructors provide the same parameter options as .NET List constructors, with the added ability to take multi-dimensional and jagged arrays as the source collection; which .NET List<T> constructors do not support. When a multi-dimensional or jagged array is used as the source collection for a UnifyList or UnifyList<T>, Unify will flatten the original source collection to populate the new Unify List.

Example One

This example demonstrates the interchangability between UnifyList<T> and .NET List<T>. A list of each type is created and populated with a list of cities where a Unity office is located. The operations; Contains, Insert, Remove and RemoveAt are used to manipulate each list and show their behaviour is identical.

Attach the script below to a GameObject and press:

  • D to run the .NET list example
  • U to run the Unify list example
using System;
using System.Collections.Generic;
using Unify.Collections.Generic;
using UnityEngine;

public class Collections_Using : MonoBehaviour
{
    private IList<string> dotnetList;
    private IUnifyList<string> unifyList;

    void Awake()
    {
        string[] cities = { "New York", "Stockholm", "Brighton", "Berlin" };
        dotnetList = new List<string>(cities);
        unifyList = new UnifyList<string>(cities);

        print("press D for .NET, U for Unify");
    }

    void Update()
    {
    	if (Input.GetKeyUp(KeyCode.D))
        {
            RunDemo(dotnetList, ".NET");
        }

        if (Input.GetKeyUp(KeyCode.U))
        {
            RunDemo(unifyList, "Unify");
        }
    }

    private void RunDemo(IList<string> list, string listType)
    {
        print($"----- {listType} Demo -----");
        PrintList(list);

        print($"Contains 'Helsinki': {list.Contains("Helsinki")}");
        print($"Contains 'Brighton': {list.Contains("Brighton")}");
        print("-------------------------");

        print("Insert(2, \"Los Angeles\")");
        list.Insert(2, "Los Angeles");
        PrintList(list);

        print("Remove(\"Stockholm\")");
        list.Remove("Stockholm");
        PrintList(list);

        print("RemoveAt(3)");
        list.RemoveAt(3);
        PrintList(list);
        print($"--------- Done ----------{Environment.NewLine}{Environment.NewLine}");
    }

    private void PrintList(IList<string> list)
    {
        for (int i = 0; i < list.Count; i++)
        {
            print(list[i]);
        }
        print("-------------------------");
    }
}

Example Two

The second example shows how two-dimensional and jagged arrays can be used to populate a UnifyList<T>.

Attach the script below to a GameObject and press:

  • A to initialise the Unify List with a two-dimensional array
  • J to initialise the Unify List with a jagged array
using System.Collections.Generic;
using Unify.Collections.Generic;
using UnityEngine;

public class Collections_Using_ExampleTwo : MonoBehaviour
{
    private IUnifyList<string> listFrom2dArray;
    private IUnifyList<string> listFromJaggedArray;

    void Awake()
    {
        listFrom2dArray = new UnifyList<string>(new string[,]
        {
            { "New York", "Austin", "Boston", "Denver" },
            { "Copenhagen", "Ghent", "Munich", "Vilnius" },
            { "Dundee", "Southampton", "Slough", "London" }
        });

        listFromJaggedArray = new UnifyList<string>(new string[][]
        {
            new string[] { "Beijing", "Bellevue", "Berlin", "Boston", "Brighton", "Burbank" }, 
            new string[] { "Ghent", "Grenoble", "Guangzhou" },
            new string[] { "Montreal", "Munich" } 
        });
 
        print("press A for 2D Array, J for Jagged Array");
    }

    void Update()
    {
        if (Input.GetKeyUp(KeyCode.A))
        {
            print("----- 2D Array -----");
            for (int i = 0; i < listFrom2dArray.Count; i++)
            {
                print($"index: {i} value: {listFrom2dArray[i]}");
            }
        }

        if (Input.GetKeyUp(KeyCode.J))
        {
            print("----- Jagged Array -----");
            for (int i = 0; i < listFromJaggedArray.Count; i++)
            {
                print($"index: {i} value: {listFromJaggedArray[i]}");
            }
        }
    }
}
